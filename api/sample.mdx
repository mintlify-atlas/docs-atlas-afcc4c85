## Overview

The `Sample` class is the core application class that orchestrates the entire RTX path tracing system. It inherits from `donut::app::ApplicationBase` and manages the rendering pipeline, scene loading, camera controls, and user interactions.

**Source:** `~/workspace/source/Rtxpt/Sample.h`, `~/workspace/source/Rtxpt/Sample.cpp`

```cpp Sample.cpp:119
Sample::Sample(donut::app::DeviceManager* deviceManager,

    : app::ApplicationBase(deviceManager), m_cmdLine(cmdLine), m_ui(ui)
```

**Key operations:**
- Creates bindless layout for GPU resources
- Initializes binding layouts for ray tracing
- Sets up constant buffers and debug buffers
- Enumerates available scene files
- Loads the preferred scene

```cpp Sample.cpp:181
bool Sample::Init(const std::string& preferredScene)
{
    // Create bindless layout
    nvrhi::BindlessLayoutDesc bindlessLayoutDesc;
    bindlessLayoutDesc.visibility = nvrhi::ShaderType::All;
    bindlessLayoutDesc.firstSlot = 0;
    bindlessLayoutDesc.maxCapacity = 1024;
    bindlessLayoutDesc.registerSpaces = {
        nvrhi::BindingLayoutItem::RawBuffer_SRV(1),
        nvrhi::BindingLayoutItem::Texture_SRV(2)
    };
    m_bindlessLayout = device->createBindlessLayout(bindlessLayoutDesc);
    
    // Create global binding layout for ray tracing
    nvrhi::BindingLayoutDesc globalBindingLayoutDesc;
    globalBindingLayoutDesc.visibility = nvrhi::ShaderType::All;
    globalBindingLayoutDesc.bindings = {
        nvrhi::BindingLayoutItem::VolatileConstantBuffer(0),
        nvrhi::BindingLayoutItem::RayTracingAccelStruct(0),
        // ... additional bindings
    };
    m_bindingLayout = device->createBindingLayout(globalBindingLayoutDesc);
    
    return true;
}
```

```cpp Sample.cpp:412
void Sample::SetCurrentScene(const std::string& sceneName, bool forceReload)
{
    if(m_currentSceneName == sceneName && !forceReload)
        return;
    m_currentSceneName = sceneName;
    m_ui.ResetAccumulation = true;
    SetAsynchronousLoadingEnabled(false);
    std::filesystem::path scenePath = GetLocalPath(c_AssetsFolder) / sceneName;
    m_currentScenePath = scenePath;
    BeginLoadingScene(std::make_shared<vfs::NativeFileSystem>(), scenePath);
}
```

```cpp Sample.cpp:459
bool Sample::LoadScene(std::shared_ptr<vfs::IFileSystem> fs, 
                       const std::filesystem::path& sceneFileName)
{
    m_scene = std::shared_ptr<ExtendedScene>(
        new ExtendedScene(GetDevice(), *m_shaderFactory, fs, m_TextureCache, 
                         m_DescriptorTable, 
                         std::make_shared<ExtendedSceneTypeFactory>()));
    if (m_scene->Load(sceneFileName))
        return true;
    m_scene = nullptr;
    return false;
}
```

```cpp Sample.cpp:1008
void Sample::CreateBlases(nvrhi::ICommandList* commandList)
{
    for (const std::shared_ptr<MeshInfo>& mesh : 
         m_scene->GetSceneGraph()->GetMeshes())
    {
        if (mesh->isSkinPrototype)
            continue;
            
        nvrhi::rt::AccelStructDesc blasDesc = 
            bvh::GetMeshBlasDesc(cfg, *mesh, nullptr, false);
        nvrhi::rt::AccelStructHandle as = GetDevice()->createAccelStruct(blasDesc);
        nvrhi::utils::BuildBottomLevelAccelStruct(commandList, as, blasDesc);
        mesh->accelStruct = as;
    }
}
```

```cpp Sample.cpp:1035
void Sample::CreateTlas(nvrhi::ICommandList* commandList)
{
    nvrhi::rt::AccelStructDesc tlasDesc;
    tlasDesc.isTopLevel = true;
    tlasDesc.topLevelMaxInstances = 
        m_scene->GetSceneGraph()->GetMeshInstances().size();
    tlasDesc.buildFlags = nvrhi::rt::AccelStructBuildFlags::PreferFastTrace;
    m_topLevelAS = GetDevice()->createAccelStruct(tlasDesc);
}
```

## Usage Example

```cpp
// Initialize the sample
Sample sample(deviceManager, cmdLineOptions, uiData);
if (!sample.Init("bistro.scene.json"))
{
    // Handle initialization failure
    return;
}

// Change scene
sample.SetCurrentScene("sponza.scene.json");

// Main render loop
while (running)
{
    sample.Animate(deltaTime);
    sample.Render(framebuffer);
}
```
